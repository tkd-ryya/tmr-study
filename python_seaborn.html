  4. 画像をパワポに自動貼り付け

    5. from pptx import Presentation

    from pptx.util import Emu

    from PIL import Image

    import os


    # 設定

    image_folder = "images" #url

    cols = 5

    rows = 10

    margin_ratio = 0.1 # 余白（10%）


    # 画像取得

    image_files = [f for f in os.listdir(image_folder) if f.lower().endswith((".jpg", ".png", ".jpeg"))]

    image_files.sort()


    # PPT作成

    prs = Presentation()

    slide = prs.slides.add_slide(prs.slide_layouts[6])

    slide_width = prs.slide_width

    slide_height = prs.slide_height


    # セルサイズ

    cell_width = slide_width / cols

    cell_height = slide_height / rows

    for idx, img_name in enumerate(image_files):

        if idx >= cols * rows:

            break

        row = idx // cols

        col = idx % cols

        img_path = os.path.join(image_folder, img_name)


        # 画像サイズ取得（ピクセル）

        with Image.open(img_path) as img:

            img_w, img_h = img.size


        # セル内で使えるサイズ（余白考慮）

        usable_w = cell_width * (1 - margin_ratio)

        usable_h = cell_height * (1 - margin_ratio)


        # 縦横比維持してスケーリング

        scale = min(usable_w / img_w, usable_h / img_h)

        new_w = img_w * scale

        new_h = img_h * scale


        # セル内中央配置

        left = col * cell_width + (cell_width - new_w) / 2

        top = row * cell_height + (cell_height - new_h) / 2

        slide.shapes.add_picture( img_path , left , top , width=new_w , height=new_h)


    # 保存

    prs.save("output_margin_keep_aspect.pptx")
